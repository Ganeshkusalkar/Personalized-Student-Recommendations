def define_student_persona(df):
    # Calculate overall metrics
    average_score = df['score'].mean()
    average_accuracy = df['accuracy'].mean()
    
    # Define persona based on performance
    if average_score >= 80:
        score_label = "High Achiever"
    elif 60 <= average_score < 80:
        score_label = "Satisfactory Performer"
    else:
        score_label = "Struggling Learner"
    
    if average_accuracy >= 0.8:
        accuracy_label = "Precision Expert"
    elif 0.6 <= average_accuracy < 0.8:
        accuracy_label = "Consistent Contributor"
    else:
        accuracy_label = "Needs Improvement"
    
    # Gather insights on preferred topics
    preferred_topic = df.groupby('quiz.topic').agg({'score': 'mean'}).reset_index()
    preferred_topic = preferred_topic.sort_values(by='score', ascending=False).head(1)

    # Construct persona summary
    persona_summary = {
        "Average Score": average_score,
        "Average Accuracy": average_accuracy,
        "Score Label": score_label,
        "Accuracy Label": accuracy_label,
        "Preferred Topic": preferred_topic['quiz.topic'].values[0],
        "Preferred Topic Score": preferred_topic['score'].values[0]
    }

    return persona_summary

# Define student persona
student_persona = define_student_persona(df)

# Display student persona
print("\nStudent Persona:")
for key, value in student_persona.items():
    print(f"{key}: {value}")

def highlight_strengths_weaknesses(persona_summary):
    strengths = []
    weaknesses = []

    # Strengths based on persona labels
    if persona_summary['Score Label'] == "High Achiever":
        strengths.append("Excellent grasp of concepts.")
    elif persona_summary['Score Label'] == "Satisfactory Performer":
        strengths.append("Good understanding of the material.")
    
    if persona_summary['Accuracy Label'] == "Precision Expert":
        strengths.append("High accuracy in responses.")
    
    # Weaknesses
    if persona_summary['Score Label'] == "Struggling Learner":
        weaknesses.append("Needs to improve understanding of key topics.")
    
    if persona_summary['Accuracy Label'] == "Needs Improvement":
        weaknesses.append("Inconsistency in answering questions accurately.")
    
    # Creative labels
    creative_strengths_labels = [f"Strength: {strength}" for strength in strengths]
    creative_weaknesses_labels = [f"Weakness: {weakness}" for weakness in weaknesses]

    return creative_strengths_labels, creative_weaknesses_labels
